# JS (script.js)
// Affiche un message de bienvenue avec l'heure actuelle
function afficherBienvenue() {
const bienvenueElement = document.getElementById('bienvenue');
if(bienvenueElement){
const maintenant = new Date();
const heures = maintenant.getHours().toString().padStart(2, '0');
const minutes = maintenant.getMinutes().toString().padStart(2, '0');
bienvenueElement.textContent = `Bienvenue sur n8n! Il est ${heures}:${minutes}`;
}
}
// Ajoute un bouton pour lancer une action et affiche un retour utilisateur
function ajouterBoutonAction() {
const container = document.getElementById('action-container');
if(container){
const btn = document.createElement('button');
btn.textContent = 'Lancer Action';
btn.onclick = () => {
alert('Action lancée avec succès !');
};
container.appendChild(btn);
}
}
// Change le thème de la page (clair/sombre) en cliquant un bouton
function basculerTheme() {
const boutonTheme = document.getElementById('btn-theme');
if(boutonTheme){
boutonTheme.addEventListener('click', () => {
document.body.classList.toggle('dark-theme');
});
}
}
// Initialisation des fonctions au chargement de la page
window.addEventListener('DOMContentLoaded', () => {
afficherBienvenue();
ajouterBoutonAction();
basculerTheme();
});
